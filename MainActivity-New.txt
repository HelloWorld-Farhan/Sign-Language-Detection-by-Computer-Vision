    // ‚úÖ NEW: Initialize Sound & Vibrate
    private void initSoundAndVibrate() {
        vibrator = (Vibrator) getSystemService(Context.VIBRATOR_SERVICE);
        try {
            toneGenerator = new ToneGenerator(AudioManager.STREAM_MUSIC, 100);
        } catch (Exception e) {
            Log.e(TAG, "ToneGenerator failed", e);
        }
    }

    // ‚úÖ NEW: Initialize Text-to-Speech
    private void initTextToSpeech() {
        textToSpeech = new TextToSpeech(this, new TextToSpeech.OnInitListener() {
            @Override
            public void onInit(int status) {
                if (status == TextToSpeech.SUCCESS) {
                    textToSpeech.setLanguage(Locale.US);
                    Log.d(TAG, "‚úÖ TTS initialized");
                }
            }
        });
    }

    // ‚úÖ NEW: Haptic feedback
    private void vibrate() {
        if (vibrator != null) {
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                vibrator.vibrate(VibrationEffect.createOneShot(50, VibrationEffect.DEFAULT_AMPLITUDE));
            } else {
                vibrator.vibrate(50);
            }
        }
    }

    // ‚úÖ NEW: Click sound
    private void playClickSound() {
        if (toneGenerator != null) {
            try {
                toneGenerator.startTone(ToneGenerator.TONE_PROP_BEEP, 100);
            } catch (Exception e) {
                Log.e(TAG, "Sound failed", e);
            }
        }
    }

    // ‚úÖ NEW: Speak text
    private void speak(String text) {
        if (textToSpeech != null && !text.isEmpty()) {
            textToSpeech.speak(text, TextToSpeech.QUEUE_FLUSH, null, null);
        }
    }

    // ‚úÖ NEW: Toggle flash
    private void toggleFlash() {
        if (camera == null) {
            Toast.makeText(this, "‚ö†Ô∏è Camera not ready", Toast.LENGTH_SHORT).show();
            return;
        }

        try {
            isFlashOn = !isFlashOn;
            camera.getCameraControl().enableTorch(isFlashOn);

            if (isFlashOn) {
                flashButton.setImageResource(R.drawable.flash);
                Toast.makeText(this, "üí° Flash ON", Toast.LENGTH_SHORT).show();
            } else {
                flashButton.setImageResource(R.drawable.flash_off);
                Toast.makeText(this, "üåô Flash OFF", Toast.LENGTH_SHORT).show();
            }

            vibrate();
            playClickSound();

        } catch (Exception e) {
            Log.e(TAG, "Flash error", e);
            Toast.makeText(this, "‚ùå Flash not available", Toast.LENGTH_SHORT).show();
        }
    }

            public void onClick(View v) {
                if (!isCapturing && canCapture) {
                    isCapturing = true;

                    Toast.makeText(MainActivity.this, "üì∏ Hold steady...", Toast.LENGTH_SHORT).show();

            @Override
            public void onClick(View v) {
                vibrate(); // ‚úÖ NEW
                playClickSound(); // ‚úÖ NEW

                    char deleted = currentWord.charAt(currentWord.length() - 1);
                    currentWord.deleteCharAt(currentWord.length() - 1);
                    wordText.setText(currentWord.toString());
                    Toast.makeText(MainActivity.this, "‚å´ Deleted: " + deleted, Toast.LENGTH_SHORT).show();
                } else {
                    Toast.makeText(MainActivity.this, "‚ö†Ô∏è Word is empty", Toast.LENGTH_SHORT).show();

        });

                        currentSentence.append(" ");

    private void saveReferenceToGallery() {
        if (Build.VERSION.SDK_INT <= Build.VERSION_CODES.P) {
            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE)
                    != PackageManager.PERMISSION_GRANTED) {
                ActivityCompat.requestPermissions(this,
                        new String[]{Manifest.permission.WRITE_EXTERNAL_STORAGE},
                        STORAGE_PERMISSION_CODE);
                return;
            }
        }

        try {
            Bitmap originalBitmap = BitmapFactory.decodeResource(getResources(), R.drawable.isl_alphabet);
            Bitmap enhancedBitmap = addWatermark(originalBitmap, "ISL Vision App");
            String fileName = "ISL_Alphabet_Reference_" + System.currentTimeMillis() + ".jpg";

            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
                ContentValues values = new ContentValues();
                values.put(MediaStore.Images.Media.DISPLAY_NAME, fileName);
                values.put(MediaStore.Images.Media.MIME_TYPE, "image/jpeg");
                values.put(MediaStore.Images.Media.RELATIVE_PATH, Environment.DIRECTORY_PICTURES + "/ISL Vision");

                Uri uri = getContentResolver().insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI, values);
                if (uri != null) {
                    OutputStream outputStream = getContentResolver().openOutputStream(uri);
                    enhancedBitmap.compress(Bitmap.CompressFormat.JPEG, 95, outputStream);
                    outputStream.close();

                    Toast.makeText(this, "‚úÖ Saved to Gallery!", Toast.LENGTH_LONG).show();
                    showSuccessDialog();
                }
            } else {
                File directory = new File(Environment.getExternalStoragePublicDirectory(
                        Environment.DIRECTORY_PICTURES), "ISL Vision");
                if (!directory.exists()) {
                    directory.mkdirs();
                }

                File file = new File(directory, fileName);
                FileOutputStream fos = new FileOutputStream(file);
                enhancedBitmap.compress(Bitmap.CompressFormat.JPEG, 95, fos);
                fos.close();

                ContentValues values = new ContentValues();
                values.put(MediaStore.Images.Media.DATA, file.getAbsolutePath());
                getContentResolver().insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI, values);

                Toast.makeText(this, "‚úÖ Saved to Pictures/ISL Vision/", Toast.LENGTH_LONG).show();
                showSuccessDialog();
            }

            Log.d(TAG, "‚úÖ Reference guide saved successfully");

        } catch (Exception e) {
            Log.e(TAG, "‚ùå Error saving reference", e);
            Toast.makeText(this, "‚ùå Error saving image: " + e.getMessage(), Toast.LENGTH_LONG).show();
        }
    }

    private Bitmap addWatermark(Bitmap src, String watermark) {
        int width = src.getWidth();
        int height = src.getHeight();

        Bitmap result = Bitmap.createBitmap(width, height, src.getConfig());
        Canvas canvas = new Canvas(result);
        canvas.drawBitmap(src, 0, 0, null);

        Paint paint = new Paint();
        paint.setColor(Color.WHITE);
        paint.setTextSize(40);
        paint.setTypeface(Typeface.create(Typeface.DEFAULT, Typeface.BOLD));
        paint.setShadowLayer(2, 2, 2, Color.BLACK);
        paint.setAntiAlias(true);

        float x = 20;
        float y = height - 30;
        canvas.drawText(watermark, x, y, paint);

        return result;
    }

    private boolean checkPermissions() {
        return ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA)
                == PackageManager.PERMISSION_GRANTED;
    }

    private void requestPermissions() {
        ActivityCompat.requestPermissions(this,
                new String[]{Manifest.permission.CAMERA},
                CAMERA_PERMISSION_CODE);
    }

    @Override
    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions,
                                           @NonNull int[] grantResults) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
        if (requestCode == CAMERA_PERMISSION_CODE) {
            if (grantResults.length > 0 && grantResults[0] == PackageManager.PERMISSION_GRANTED) {
                startCamera();
            } else {
                Toast.makeText(this, "‚ùå Camera permission required", Toast.LENGTH_LONG).show();
                finish();
            }
        } else if (requestCode == STORAGE_PERMISSION_CODE) {
            if (grantResults.length > 0 && grantResults[0] == PackageManager.PERMISSION_GRANTED) {
                saveReferenceToGallery();
            } else {
                Toast.makeText(this, "‚ö†Ô∏è Storage permission needed to save image", Toast.LENGTH_LONG).show();
            }
        }
    }

    private void startCamera() {
        ListenableFuture<ProcessCameraProvider> cameraProviderFuture =
                ProcessCameraProvider.getInstance(this);

        cameraProviderFuture.addListener(() -> {
            try {
                ProcessCameraProvider cameraProvider = cameraProviderFuture.get();
                bindCameraUseCases(cameraProvider);
            } catch (ExecutionException | InterruptedException e) {
                Log.e(TAG, "‚ùå Error starting camera", e);
            }
        }, ContextCompat.getMainExecutor(this));
    }

    private void bindCameraUseCases(ProcessCameraProvider cameraProvider) {
        Preview preview = new Preview.Builder()
                .setTargetAspectRatio(AspectRatio.RATIO_4_3)
                .build();

        preview.setSurfaceProvider(previewView.getSurfaceProvider());

        ImageAnalysis imageAnalysis = new ImageAnalysis.Builder()
                .setTargetAspectRatio(AspectRatio.RATIO_4_3)
                .setBackpressureStrategy(ImageAnalysis.STRATEGY_KEEP_ONLY_LATEST)
                .setOutputImageFormat(ImageAnalysis.OUTPUT_IMAGE_FORMAT_YUV_420_888)
                .build();

        imageAnalysis.setAnalyzer(cameraExecutor, this::analyzeImage);

        CameraSelector cameraSelector = new CameraSelector.Builder()
                .requireLensFacing(CameraSelector.LENS_FACING_BACK)
                .build();

        try {
            cameraProvider.unbindAll();
            camera = cameraProvider.bindToLifecycle(this, cameraSelector, preview, imageAnalysis); // ‚úÖ Store camera
            Log.d(TAG, "‚úÖ Camera initialized");
        } catch (Exception e) {
            Log.e(TAG, "‚ùå Camera binding failed", e);
        }
    }

    private void analyzeImage(ImageProxy imageProxy) {
        if (!isCapturing) {
            imageProxy.close();
            return;
        }

        try {
            Bitmap bitmap = imageProxyToBitmap(imageProxy);
            if (bitmap != null) {
                Bitmap croppedBitmap = cropToCenter(bitmap);
                SignLanguageClassifier.Result result = classifier.classify(croppedBitmap);
                runOnUiThread(() -> processResult(result));
                croppedBitmap.recycle();
            }
        } catch (Exception e) {
            Log.e(TAG, "‚ùå Error analyzing image", e);
        } finally {
            imageProxy.close();
        }
    }

    private Bitmap cropToCenter(Bitmap source) {
        int size = Math.min(source.getWidth(), source.getHeight());
        int x = (source.getWidth() - size) / 2;
        int y = (source.getHeight() - size) / 2;

        Bitmap cropped = Bitmap.createBitmap(source, x, y, size, size);
        if (cropped != source) {
            source.recycle();
        }
        return cropped;
    }

    private Bitmap imageProxyToBitmap(ImageProxy image) {
        ImageProxy.PlaneProxy[] planes = image.getPlanes();
        ByteBuffer yBuffer = planes[0].getBuffer();
        ByteBuffer uBuffer = planes[1].getBuffer();
        ByteBuffer vBuffer = planes[2].getBuffer();

        int ySize = yBuffer.remaining();
        int uSize = uBuffer.remaining();
        int vSize = vBuffer.remaining();

        byte[] nv21 = new byte[ySize + uSize + vSize];
        yBuffer.get(nv21, 0, ySize);
        vBuffer.get(nv21, ySize, vSize);
        uBuffer.get(nv21, ySize + vSize, uSize);

        YuvImage yuvImage = new YuvImage(nv21, ImageFormat.NV21,
                image.getWidth(), image.getHeight(), null);
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        yuvImage.compressToJpeg(new Rect(0, 0, image.getWidth(), image.getHeight()), 100, out);
        byte[] imageBytes = out.toByteArray();

        Bitmap bitmap = BitmapFactory.decodeByteArray(imageBytes, 0, imageBytes.length);

        Matrix matrix = new Matrix();
        matrix.postRotate(90);
        Bitmap rotatedBitmap = Bitmap.createBitmap(bitmap, 0, 0,
                bitmap.getWidth(), bitmap.getHeight(), matrix, true);
        if (rotatedBitmap != bitmap) {
            bitmap.recycle();
        }

        return rotatedBitmap;
    }

    private void processResult(SignLanguageClassifier.Result result) {
        resultText.setText(result.getLabel());
        confidenceText.setText("Confidence: " + result.getConfidencePercent() + "%");

        if (result.getConfidence() >= CONFIDENCE_THRESHOLD) {
            confidenceText.setTextColor(0xFF4CAF50);

            if (result.getLabel().equals(lastPredictedLetter)) {
                consecutiveCount++;
            } else {
                lastPredictedLetter = result.getLabel();
                consecutiveCount = 1;
            }

            if (consecutiveCount >= CONFIRMATION_FRAMES) {
                addLetterToWord(result.getLabel());
                resetCapture();
            }
        } else {
            confidenceText.setTextColor(0xFFFF5722);
        }
    }

    private void addLetterToWord(String letter) {
        currentWord.append(letter);
        wordText.setText(currentWord.toString());

        captureButton.setBackgroundTintList(
                android.content.res.ColorStateList.valueOf(0xFF4CAF50));

        Toast.makeText(this, "‚úÖ Added: " + letter, Toast.LENGTH_SHORT).show();

        // ‚úÖ NEW: Success feedback
        vibrate();
        playClickSound();

        canCapture = false;
        mainHandler.postDelayed(() -> {
            canCapture = true;
            captureButton.setBackgroundTintList(
                    android.content.res.ColorStateList.valueOf(0xFF4CAF50));
        }, COOLDOWN_MS);
    }

    private void resetCapture() {
        isCapturing = false;
        lastPredictedLetter = "";
        consecutiveCount = 0;
        confidenceText.setTextColor(0xFF666666);
    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
        if (classifier != null) {
            classifier.close();
        }
        if (cameraExecutor != null) {
            cameraExecutor.shutdown();
        }
        // ‚úÖ NEW: Cleanup
        if (textToSpeech != null) {
            textToSpeech.stop();
            textToSpeech.shutdown();
        }
        if (toneGenerator != null) {
            toneGenerator.release();
        }
        Log.d(TAG, "üõë ISL Vision stopped");
    }
